class Solution {
    public boolean isAnagram(String s, String t) {
        // Step 1: If lengths differ, cannot be anagrams
        if (s.length() != t.length()) {
            return false;
        }

        // Step 2: Frequency count for each character
        int[] count = new int[26];

        // Step 3: Count characters in s and t simultaneously
        for (int i = 0; i < s.length(); i++) {
            count[s.charAt(i) - 'a']++;
            count[t.charAt(i) - 'a']--;
        }

        // Step 4: Verify all counts are zero
        for (int c : count) {
            if (c != 0) return false;
        }

        return true;
    }
}

import java.util.*;

class Solution {
    public List<Integer> grayCode(int n) {
        List<Integer> result = new ArrayList<>();
        int totalNumbers = 1 << n; // 2^n numbers

        for (int i = 0; i < totalNumbers; i++) {
            result.add(i ^ (i >> 1)); // Formula for Gray Code
        }

        return result;
    }
}
